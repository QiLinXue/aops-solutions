\section{Equations}

There are 6 main equations:
\begin{equation}
    \log_a{b^n} = n\log_a{b}
\end{equation}
\begin{equation}
    \log_a{b} + \log_a{c} = \log_a{bc}
\end{equation}
\begin{equation}
    \log_a{b} - \log_a{a} = \log_a{b/c}
\end{equation}
\begin{equation}
    (\log_a{b})(\log_c{d}) = (\log_a{d})(\log_c{b})
\end{equation}
\begin{equation}
    \frac{\log_a{b}}{\log_a{c}} = \log_c{b}
\end{equation}
\begin{equation}
    \log_{a^n}{b^n} = \log_a{b}
\end{equation}
